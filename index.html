1 El Contenedor HTML

El motor es agnóstico al framework (funciona en React, Vue, o Vanilla). Solo necesita un Canvas dentro de un contenedor con dimensiones definidas.

<!-- El canvas tomará el 100% del width/height de este contenedor -->
<div id="mi-contenedor" style="width: 100%; height: 800px;">
    <canvas id="mi-lienzo"></canvas>
</div>

2 El Motor (Core Class V3)

Este es el cerebro del renderizado. Contiene el caché de memoria, el escáner de canal alfa y el algoritmo de "Chunks" para la justificación. Cópialo en tu archivo JS principal.

Librería: EditorialCanvasV3 
class EditorialCanvasV3 {
    constructor(canvasId, opciones) {
        this.canvas = document.getElementById(canvasId);
        this.ctx = this.canvas.getContext('2d', { alpha: false }); 
        this.config = { ...opciones };
        this.palabras = [];
        this.wordCache = new Map();
        this.imagenCargada = false;
        this.imagenObj = new Image();
        this.mapaSegmentos = []; 
        this.prepararTexto();
        if(this.config.imgSrc) this.cargarImagen(this.config.imgSrc);
        this.resizeTimer = null;
        window.addEventListener('resize', () => {
            clearTimeout(this.resizeTimer);
            this.resizeTimer = setTimeout(() => this.ajustarResolucion(), 100);
        });
        this.ajustarResolucion();
    }

    prepararTexto() {
        this.ctx.font = this.config.fuente;
        const rawWords = this.config.texto.split(/\s+/).filter(p => p.length > 0);
        this.palabras = [];
        rawWords.forEach(word => {
            if (!this.wordCache.has(word)) this.wordCache.set(word, this.ctx.measureText(word).width);
            this.palabras.push({ texto: word, ancho: this.wordCache.get(word) });
        });
        this.anchoEspacio = this.ctx.measureText(" ").width;
    }

    actualizarConfig(nuevasOpciones) {
        let recargarImg = nuevasOpciones.imgSrc && nuevasOpciones.imgSrc !== this.config.imgSrc;
        let recalcularMapa = recargarImg || nuevasOpciones.exclusionRadio !== undefined;
        let recalcularTexto = nuevasOpciones.texto !== undefined || nuevasOpciones.fuente !== undefined;
        this.config = { ...this.config, ...nuevasOpciones };
        if (recalcularTexto) this.prepararTexto();
        if (recargarImg) this.cargarImagen(this.config.imgSrc);
        else if (recalcularMapa && this.imagenCargada) this.procesarMapaSDF();
        this.renderizar();
    }

    cargarImagen(src) {
        this.imagenCargada = false;
        this.imagenObj.onload = () => { this.imagenCargada = true; this.procesarMapaSDF(); this.renderizar(); };
        this.imagenObj.src = src;
    }

    procesarMapaSDF() {
        if(!this.imagenCargada) return;
        const dimension = this.config.exclusionRadio * 2;
        const ratio = this.imagenObj.width / this.imagenObj.height;
        let imgW = dimension, imgH = dimension;
        if(ratio > 1) imgH = dimension / ratio; else imgW = dimension * ratio;
        this.imgW = Math.floor(imgW); this.imgH = Math.floor(imgH);

        const offCanvas = document.createElement('canvas');
        offCanvas.width = this.imgW; offCanvas.height = this.imgH;
        const offCtx = offCanvas.getContext('2d', { willReadFrequently: true });
        offCtx.drawImage(this.imagenObj, 0, 0, this.imgW, this.imgH);
        const data = offCtx.getImageData(0, 0, this.imgW, this.imgH).data;
        this.mapaSegmentos = [];

        for (let y = 0; y < this.imgH; y++) {
            let segmentosFila = [], enSolido = false, inicioX = 0;
            for (let x = 0; x < this.imgW; x++) {
                const esOpaco = data[(y * this.imgW + x) * 4 + 3] > 15; 
                if (esOpaco && !enSolido) { enSolido = true; inicioX = x; } 
                else if (!esOpaco && enSolido) { enSolido = false; segmentosFila.push({ left: inicioX, right: x }); }
            }
            if (enSolido) segmentosFila.push({ left: inicioX, right: this.imgW });
            this.mapaSegmentos[y] = segmentosFila.length > 0 ? segmentosFila : null;
        }
    }

    ajustarResolucion() {
        const dpr = window.devicePixelRatio || 1;
        const rect = this.canvas.parentElement.getBoundingClientRect();
        this.canvas.width = rect.width * dpr; this.canvas.height = rect.height * dpr;
        this.ctx.scale(dpr, dpr);
        this.width = rect.width; this.height = rect.height;
        this.anchoMaximo = this.width - (this.config.paddingX * 2);
        this.prepararTexto(); this.renderizar();
    }

    renderizar() {
        this.ctx.fillStyle = this.config.bgCanvas || '#fff';
        this.ctx.fillRect(0, 0, this.width, this.height);
        
        if(this.config.tipoForma === 'circulo') {
            this.ctx.beginPath();
            this.ctx.arc(this.config.exX, this.config.exY, this.config.exclusionRadio, 0, Math.PI * 2);
            this.ctx.fillStyle = this.config.colorExclusion || '#3b82f6';
            this.ctx.fill();
        } else if (this.imagenCargada) {
            this.ctx.drawImage(this.imagenObj, this.config.exX - (this.imgW/2), this.config.exY - (this.imgH/2), this.imgW, this.imgH);
        }

        this.ctx.font = this.config.fuente; this.ctx.fillStyle = this.config.colorTexto; this.ctx.textBaseline = 'middle';

        let cursorX = this.config.paddingX, cursorY = this.config.paddingY;
        const rep = this.config.margenExclusion;
        let chunk = [], chunkStartX = cursorX;

        const flushChunk = (endX, justificar) => {
            if (chunk.length === 0) return;
            let totalWordsWidth = chunk.reduce((sum, p) => sum + p.ancho, 0);
            let numEspacios = chunk.length - 1, espacioExtra = this.anchoEspacio;
            
            if (this.config.justificado && justificar && numEspacios > 0) {
                let calcEspacio = (endX - chunkStartX - totalWordsWidth) / numEspacios;
                if (calcEspacio <= this.anchoEspacio * 4) espacioExtra = calcEspacio; 
            }
            let drawX = chunkStartX;
            for (let p of chunk) { this.ctx.fillText(p.texto, drawX, cursorY); drawX += p.ancho + espacioExtra; }
            chunk = []; 
        };

        for (let i = 0; i < this.palabras.length; i++) {
            const objPalabra = this.palabras[i];
            while (true) {
                if (cursorX + objPalabra.ancho > this.width - this.config.paddingX) {
                    flushChunk(this.width - this.config.paddingX, true); 
                    cursorX = this.config.paddingX; cursorY += this.config.altoLinea; chunkStartX = cursorX;
                }
                if (cursorY > this.height + this.config.altoLinea) return; 

                let huboColision = false, colBordeIzq = 0, colBordeDer = 0;

                if (this.config.tipoForma === 'circulo') {
                    let distY = cursorY - this.config.exY, radioEfectivo = this.config.exclusionRadio + rep;
                    if (Math.abs(distY) < radioEfectivo) {
                        let distX = Math.sqrt((radioEfectivo ** 2) - (distY ** 2));
                        let bIzq = this.config.exX - distX, bDer = this.config.exX + distX;
                        if (cursorX + objPalabra.ancho > bIzq && cursorX < bDer) { huboColision = true; colBordeIzq = bIzq; colBordeDer = bDer; }
                    }
                } else if (this.imagenCargada) {
                    let localY = Math.floor(cursorY - (this.config.exY - (this.imgH/2)));
                    if(localY >= 0 && localY < this.imgH && this.mapaSegmentos[localY]) {
                        let imgLeftEdge = this.config.exX - (this.imgW/2);
                        for(let seg of this.mapaSegmentos[localY]) {
                            let bIzq = imgLeftEdge + seg.left - rep, bDer = imgLeftEdge + seg.right + rep;
                            if (cursorX + objPalabra.ancho > bIzq && cursorX < bDer) { huboColision = true; colBordeIzq = bIzq; colBordeDer = bDer; break; }
                        }
                    }
                }

                if (huboColision) {
                    flushChunk(colBordeIzq, true); 
                    cursorX = colBordeDer; chunkStartX = cursorX; 
                    continue; 
                }
                chunk.push(objPalabra); cursorX += objPalabra.ancho + this.anchoEspacio; break;
            }
        }
        flushChunk(this.width - this.config.paddingX, false);
    }
}
